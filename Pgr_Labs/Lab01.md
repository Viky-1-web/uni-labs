```java
import java.util.HashMap; //–±–∏–±–ª–∏–æ—Ç–µ–∫–∞ –¥–∂–∞–≤—ã
import java.util.Map;

public class Leetcode { // –Ω–∞–∑–≤–∞–Ω–∏–µ –∫–ª–∞—Å—Å–∞
    public static void main(String[] args) {
        // —Å–æ–∑–¥–∞–Ω–∏–µ –º–∞—Å—Å–∏–≤–∞ –∏ –ø—Ä–∏—Å–≤–æ–µ–Ω–∏–µ –µ–º—É –∑–Ω–∞—á–µ–Ω–∏–π
        Object[] arr = { true, "hello", 5, 12, -200, false, false, "word" };
        // –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏ HashMap
        Map<String, Integer> type = new HashMap<>();

        // —Ü–∏–∫–ª "for-each"
        // –ø–µ—Ä–µ–±–∏—Ä–∞–µ—Ç –∑–Ω–∞—á–µ–Ω–∏—è –º–∞—Å—Å–∏–≤–∞ arr –∏ –∫–∞–∂–¥—ã–π –µ–ª–µ–º–µ–Ω—Ç –ø—Ä–∏—Å–≤–∞–∏–≤–∞–µ—Ç elem
        for (Object elem : arr) {
            String name; // –≤ –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é –≤–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è –Ω–∞–≤–∞–Ω–∏–µ –µ–ª–µ–º–µ–Ω—Ç–∞

            // —Ü–∏–∫–ª "if-else"
            if (elem == null) { //
                name = "null";
            } else {
                name = elem.getClass().getSimpleName();
            }

            // –ø–æ–¥—Å—á—ë—Ç –∫–ª—é—á (String/Boolean) —Å –±–∏–±–ª–∏–æ—Ç–µ–∫–∏ –∏ –¥–æ–±–∞–≤–ª—è–µ—Ç +1 (—É–≤–µ–ª–∏—á–∏–≤–∞–µ—Ç
            // —Å—á—ë—Ç—á–∏–∫ —Ç–∏–ø–∞)
            type.put(name, type.getOrDefault(name, 0) + 1);
            // type.put -> –æ–±–Ω–æ–≤–ª—è–µ—Ç –∏ –¥–æ–±–∞–≤–ª—è–µ—Ç –∑–Ω–∞—á–µ–Ω–∏–µ –≤ Map
        }

        // –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –ª—è–º–±–∞
        // –º–µ—Ç–æ–¥ "for-each", –∫–æ—Ç–æ—Ä—ã–π –ø–æ–¥–±–∏—Ä–∞–µ—Ç –Ω—É–∂–Ω—ã–µ –∫–ª—é—á–∏ –∏ –∑–Ω–∞—á–µ–Ω–∏—è, –ø–æ—Å–ª–µ —á–µ–≥–æ
        // –≤—ã–≤–æ–¥–∏—Ç
        type.forEach((key, value) -> System.out.println(key + ": " + value));

    }
}
```
